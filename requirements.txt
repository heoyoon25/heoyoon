def add_two_numbers(a, b):
    return a + b

def sum_up_to_n(n):
    total = 0
    for i in range(1, n + 1):
        total += i
    return total

try:
    A = int(input("첫 번째 숫자 A를 입력하세요: "))
    B = int(input("두 번째 숫자 B를 입력하세요: "))
except ValueError:
    print("유효한 숫자를 입력해 주세요.")
    exit()

sum_of_A_and_B = add_two_numbers(A, B)
print(f"A와 B의 합은 {sum_of_A_and_B}입니다.")

print(f"\n1. 숫자 A ({A})를 선택")
print(f"2. 숫자 B ({B})를 선택")

try:
    choice = int(input("어떤 숫자를 선택하시겠습니까? (1 또는 2를 입력): "))

    if choice == 1:
        selected_number = A
    elif choice == 2:
        selected_number = B
    else:
        print("잘못된 선택입니다. 1 또는 2만 입력해 주세요.")
        exit()
except ValueError:
    print("유효한 숫자를 입력해 주세요.")
    exit()

sum_of_range = sum_up_to_n(selected_number)
print(f"선택된 숫자 ({selected_number})까지의 모든 수의 합은 {sum_of_range}입니다.")
